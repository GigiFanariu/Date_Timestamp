CREATE VIEW MEAL_ORDER_WITH_DETAILS AS 
    SELECT MO.*, M.PRICE, M.CALORIES FROM MEAL_ORDER MO, MEAL M 
    WHERE MO.MEAL_NAME = M.NAME;

SELECT
CONCAT('Plate With=', MEAL_NAME) AS NAME_FORMATED,
LOWER(MEAL_NAME) AS LOWER_CASE,
UPPER(MEAL_NAME) AS UPPERCASE_NAME,
TRIM(MEAL_NAME) AS TRIMED_NAME,
LENGTH(MEAL_NAME) AS LENGTH_NAME,
SUBSTR(MEAL_NAME, 5) AS SUBSTRING_NAME,
ROUND(3.14159, 2) AS ROUND_NUMBER,
DATE_ADD(TIME_MEAL_ORDERED, INTERVAL 1 DAY) AS DATE_ADD,
DATE_SUB(TIME_MEAL_ORDERED, INTERVAL 1 YEAR) AS DATE_SUB,
FORMAT(NOW(), 'STRING') AS FORMAT_STRING,
DATE_ADD(DATE_ADD(NOW(), INTERVAL 7 DAY), INTERVAL 11 MONTH) AS BIRTHDAY_DATE
FROM MEAL_ORDER_WITH_DETAILS;
